< // register events canvas.onmousedown = handleOnMouseDown; canvas.onmouseup = handleOnMouseUp; canvas.onmouseout = handleOnMouseOut; canvas.onmousemove = handleOnMouseMove; function handleOnMouseDown(e) { // remove old selected if (selectedBlock != null) { imageBlockList[selectedBlock.no].isSelected = false; } selectedBlock = GetImageBlock(imageBlockList, e.pageX, e.pageY); if (selectedBlock) { imageBlockList[selectedBlock.no].isSelected = true; } }
